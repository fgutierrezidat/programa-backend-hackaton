Algoritmo ContarNumeros
	Definir i, numero, contador Como Entero
	Definir numerosAlmacenados Como Cadena 
	Para i = 1 Hasta 5 Con Paso 1 Hacer
		Escribir "Ingresar numero ", i, " :"
		Leer numero
		Si numero > 50 Entonces
			contador = contador + 1
			Si contador > 1 Entonces 
				numerosAlmacenados = numerosAlmacenados + ", " + ConvertirATexto(numero)
			SiNo
				numerosAlmacenados = ConvertirATexto(numero)
			FinSi
		Fin Si
	Fin Para
	
	Escribir "Cantidad de números mayores a 50 ingresados: ", contador
	Si contador > 0 Entonces
		Escribir "Los números mayores a 50 son: ", numerosAlmacenados
	FinSi
	
FinAlgoritmo
