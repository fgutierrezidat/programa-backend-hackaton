Algoritmo sin_titulo
	Definir user, password Como Cadena;
	Definir lUser, lPassword, conNumero Como Entero;
	Definir i Como Entero;
	Definir carac Como Caracter;
	
    Escribir "Ingrese el nombre de usuario:";
    Leer user;
    Escribir "Ingrese la contraseña:";
    Leer password;
	
	lUser <- Longitud(user);
	lPassword <- Longitud(password);
	
	conNumero <- 0;
	
	Para i<-1 Hasta lPassword Hacer
		carac <- Subcadena(password,i,i)
		Si carac >= "0" y carac <= "9" Entonces
			conNumero <- 1;
		FinSi
	FinPara
	
	Si lUser >= 6 y lPassword >= 8 y conNumero = 1 Entonces
		Escribir "Acceso concedido"
	SiNo
		Si lUser < 6 Entonces
			Escribir "El nombre de usuario debe tener minimo 6 caracteres.";
		FinSi
		
		Si lPassword < 8 Entonces
			Escribir "La contraseña debe tener minimo 8 caracteres.";
		FinSi
		
		Si conNumero = 0 Entonces
			Escribir "La contraseña debe contener al menos un caracter numerico.";
		FinSi
	FinSi
	
FinAlgoritmo
