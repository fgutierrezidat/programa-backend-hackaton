Algoritmo sin_titulo
	definir num, numinv, a, b, c, d como entero;
	escribir "Ingrese el número a invertir:";
	Leer num;
	
	a <- trunc (num/1000);
	b <- trunc ((num mod 1000)/100);
	c <- trunc ((num mod 100)/10);
	d <- num mod 10;
	numinv <- d * 1000 + c * 100 + b * 10 + a;
	
	Escribir "El número invertido es:", numinv;
FinAlgoritmo
