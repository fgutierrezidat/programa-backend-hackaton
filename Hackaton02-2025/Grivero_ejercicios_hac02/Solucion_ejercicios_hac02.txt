Ejercicio 8
Algoritmo Validacion_de_credenciales
	Definir usuario, contrasena como caracter
	definir i como entero
	definir tieneNumero Como Logico
	definir usuarioValido Como Logico
	definir contrasenaValido Como Logico
	
	tieneNumero <- Falso
	
	Escribir "Ingrese un usuario"
	leer usuario
	
	Escribir "Ingrese contrasena"
	leer contrasena
	
	para i <- 1 hasta longitud(contrasena) Hacer
		si Subcadena(contrasena, i, i) >= "0" y Subcadena(contrasena, i, i) <= "9" Entonces
			tieneNumero <- verdadero
		FinSi
	FinPara
	
	usarioValido <- longitud(usuario) >= 6
	contrasenaValido <- longitud(contrasena) >= 8
	
	si usuarioValido y contrasenaValido y tieneNumero Entonces
		escribir "acceso concedido"
	sino 
		si !usuarioValido Entonces
			Escribir "El usuario debe tener al menos 6 caracteres"
		FinSi
		si !contrasenaValido entonces
			escribir "La contrasena debe tener al menos 8 caracteres"
		FinSi
		si !tieneNumero entonces
			escribir "La contrasena debe contener al menos un numero"
		FinSi
	FinSi
FinAlgoritmo
